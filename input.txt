func verificarSigno(n) {
    if (n < 0) {
        print("El número es negativo");
        return -1;
    } else if (n > 0) {
        print("El número es positivo");
        return 1;
    } else {
        print("El número es cero");
        return 0;
    }
}

print(verificarSigno(-5));
print(verificarSigno(3));
print(verificarSigno(0));
